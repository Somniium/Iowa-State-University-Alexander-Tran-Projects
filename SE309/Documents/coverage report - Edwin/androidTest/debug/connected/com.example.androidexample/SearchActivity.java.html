    /**
     * This helper method searches our database for saved books. This keeps lookup times short
     * for items that have already been searched for. If the book is found, this method will
     * update UI elements. If not, it calls the helper method bookSearchAPIAndSave to find the
     * book in an external database.
     *
     * @param mediaTitle The title of the book to be searched for.
     */
    private void bookSearchRepository(String mediaTitle) {
        url = String.format("http://coms-3090-017.class.las.iastate.edu:8080/book/name?title=%s", mediaTitle);
        JsonObjectRequest saveBookByVolumeId = new JsonObjectRequest(Request.Method.GET, // HTTP method (GET request)
                url, // API URL
                null, // Request body (null for GET request)
                response -> {
                    //extract information out of returned JSON book object
                    String title;
                    String author;
                    String year;
                    String volumeID;
                    try {
                        Log.d("Server response:", response.get("title").toString());
                        title = response.get("title").toString();
                        author = response.get("authors").toString();
                        year = response.get("publishedDate").toString();
                        volumeID = response.get("volumeId").toString();
                    } catch (JSONException e) {
                        throw new RuntimeException(e);
                    }
                    //display results page for books
                    bookResult.setVisibility(View.VISIBLE);
                    bookHeader.setText("Book Found!");
                    bookTitle.setText("Title: " + title);
                    bookAuthor.setText("Author: " + author);
                    bookDate.setText("Year Published: " + year);
                    bookButton.setText("View Book");

                    nameOfNeededItem = title;
                    nameOfNeededAuthor = author;
                },
                error -> {
                    int errorCode = error.networkResponse.statusCode;

                    if (errorCode == 404) { //If book could not be found in repository (404 error), use API search (OpenLibrary)
                        Log.d("Search: ", "Book not found in repository, forwarding to API...");
                        //format mediaTitle to work in an APi queryparam
                        String fixedTitle = mediaTitle.replace(" ", "_");
                        bookSearchAPIAndSave(fixedTitle);
                    } else {
                        Toast.makeText(getApplicationContext(), "Unknown error.", Toast.LENGTH_LONG).show();
                    }

                }
        ) {
            @Override
            public Map<String, String> getHeaders() throws AuthFailureError {
                // Define headers if needed
                HashMap<String, String> headers = new HashMap<>();
                return headers;
            }

            @Override
            protected Map<String, String> getParams() {
                // Define parameters if needed
                Map<String, String> params = new HashMap<>();
                // Example parameters (uncomment if needed)
                // params.put("param1", "value1");
                // params.put("param2", "value2");
                return params;
            }
        };

        // Adding request to the Volley request queue
        VolleySingleton.getInstance(getApplicationContext()).addToRequestQueue(saveBookByVolumeId);
    }

    /**
     * This helper method is called if the book cannot be found in our database. This method will
     * search an external database (currently, OpenLibrary) to try to find details on the book. Those
     * details will then be saved and send to our database to log the book.
     * <p>
     * This method returns nothing, but will call the helper method bookSave.
     *
     * @param mediaTitle The title of the book to be searched for.
     */
    private void bookSearchAPIAndSave(String mediaTitle) {
        url = String.format("http://coms-3090-017.class.las.iastate.edu:8080/search-book?q=%s", mediaTitle);

        JsonArrayRequest databaseRequest = new JsonArrayRequest(Request.Method.GET, // HTTP method (GET request)
                url, // API URL
                null, // Request body (null for GET request)
                new Response.Listener<JSONArray>() {
                    @Override
                    public void onResponse(JSONArray response) {
                        Log.d("Test:", response.toString());

                        if (response.length() == 0) {
                            Toast.makeText(getApplicationContext(), "Book \"" + mediaTitle + "\"could not be found.", Toast.LENGTH_LONG).show();
                            return;
                        }
                        JSONObject item = null;
                        try {
                            item = response.getJSONObject(0);
                        } catch (JSONException e) {
                            throw new RuntimeException(e);
                        }

                        //extract information out of returned JSON book object
                        String title;
                        String author;
                        String year;
                        String volumeID;
                        try {
                            Log.d("Server response:", item.get("title").toString());
                            title = item.get("title").toString();
                            author = item.get("authors").toString();
                            year = item.get("publishedDate").toString();
                            volumeID = item.get("volumeId").toString();
                        } catch (JSONException e) {
                            throw new RuntimeException(e);
                        }

                        //save book to database by volumeID for faster lookup next time
                        bookSave(volumeID);

                        //display results page for books
                        bookResult.setVisibility(View.VISIBLE);
                        bookHeader.setText("Book Found!");
                        bookTitle.setText("Title: " + title);
                        bookAuthor.setText("Author: " + author);
                        bookDate.setText("Year Published: " + year);
                        bookButton.setText("View Book");

                        nameOfNeededItem = title;
                        nameOfNeededAuthor = author;

                    }
                },
                new Response.ErrorListener() {
                    @Override
                    public void onErrorResponse(VolleyError error) {
                        int errorCode = error.networkResponse.statusCode;

                        if (errorCode == 404) {
                            Toast.makeText(getApplicationContext(), "Book \"" + mediaTitle + "\"could not be found.", Toast.LENGTH_LONG).show();
                        } else {
                            Toast.makeText(getApplicationContext(), "Unknown error.", Toast.LENGTH_LONG).show();
                        }
                    }
                }
        ) {
            @Override
            public Map<String, String> getHeaders() throws AuthFailureError {
                // Define headers if needed
                HashMap<String, String> headers = new HashMap<>();
                return headers;
            }

            @Override
            protected Map<String, String> getParams() {
                // Define parameters if needed
                Map<String, String> params = new HashMap<>();
                // Example parameters (uncomment if needed)
                // params.put("param1", "value1");
                // params.put("param2", "value2");
                return params;
            }
        };

        // Adding request to the Volley request queue
        VolleySingleton.getInstance(getApplicationContext()).addToRequestQueue(databaseRequest);
    }
