                if (!query.isEmpty()) {
                    filterLocalResults(query);
                } else {
                    // Empty search — reset to chip filter
                    filterByChip();

                return true;

            return false;

            if (reviewPhase.getVisibility() == View.VISIBLE) {
                showSearchPhase();

    private synchronized void checkMediaLoaded() {

    // ==================== LOAD ALL MEDIA ====================

    private void filterByChip() {
        searchResults.clear();
        if (selectedChip.equals("ALL")) {
            searchResults.addAll(allMediaResults);
        } else {
            for (DTOmodels.SearchResult item : allMediaResults) {
                if (selectedChip.equals(item.mediaType)) {
                    searchResults.add(item);
                }
            }
        }
        resultAdapter.notifyDataSetChanged();
        Log.d("CreatePost", "Showing " + searchResults.size() + " results for chip: " + selectedChip);
    }

    private void filterLocalResults(String query) {
        searchResults.clear();
        String lower = query.toLowerCase();

        // Filter from allMediaResults, respecting current chip
        for (DTOmodels.SearchResult item : allMediaResults) {
            if (!selectedChip.equals("ALL") && !selectedChip.equals(item.mediaType)) continue;
            if (item.title != null && item.title.toLowerCase().contains(lower)) {
                searchResults.add(item);
            }
        }
        resultAdapter.notifyDataSetChanged();

        // Always search external APIs too, results will be added alongside local ones
        Log.d("CreatePost", "Local results: " + searchResults.size() + ", also searching APIs...");
        searchMedia(query);
    }

    // ==================== FILTERING ====================

    private void searchMedia(String query) {
        pendingSearches = 0;

        if (selectedChip.equals("ALL") || selectedChip.equals("MUSIC")) {
            pendingSearches++;
            searchAlbums(query);
        }
        if (selectedChip.equals("ALL") || selectedChip.equals("GAME")) {
            pendingSearches++;
            searchGames(query);
        }
        if (selectedChip.equals("ALL") || selectedChip.equals("MOVIE")) {
            pendingSearches++;
            searchMovies(query);
        }
        if (selectedChip.equals("ALL") || selectedChip.equals("BOOK")) {
            pendingSearches++;
            searchBooks(query);
        }
        if (selectedChip.equals("ALL") || selectedChip.equals("ARTIST")) {
            pendingSearches++;
            searchArtist(query);
        }
    }

    private void searchAlbums(String query) {
        ApiClient.get(getContext(), "/search-albums-and-save?album_name=" + query, new Api_Interface() {
            @Override
            public void onSuccess(JSONObject response) {
                try {
                    DTOmodels.album album = new Gson().fromJson(response.toString(), DTOmodels.album.class);
                    addSearchResult(new DTOmodels.SearchResult(
                            album.albumId, album.name, album.genre, "MUSIC", album.coverURL));
                    Log.d("CreatePost", "Album found via API: " + album.name);
                } catch (Exception e) {
                    Log.e("CreatePost", "Album parse error: " + e.getMessage());
                }
                checkSearchDone();
            }

            @Override
            public void onError(String message) {
                Log.e("CreatePost", "Album search failed: " + message);
                checkSearchDone();
            }
        });
    }

    private synchronized void checkSearchDone() {
        pendingSearches--;
        if (pendingSearches <= 0 && searchResults.isEmpty() && getContext() != null) {
            Toast.makeText(getContext(), "No results found", Toast.LENGTH_SHORT).show();
        }
    }

    private void addSearchResult(DTOmodels.SearchResult result) {
        for (DTOmodels.SearchResult existing : searchResults) {
            // Skip if same title AND same type
            if (existing.title.equals(result.title) && existing.mediaType.equals(result.mediaType)) {
                return;
            }
            // Skip if same external ID
            if (existing.externalId != null && result.externalId != null
                    && existing.externalId.equals(result.externalId)) {
                return;
            }
        }
        searchResults.add(result);
        resultAdapter.notifyDataSetChanged();
    }

    private void publish() {
        String ratingStr = ratingInput.getText().toString().trim();
        String body = bodyInput.getText().toString().trim();
        String caption = captionInput.getText().toString().trim();

        if (ratingStr.isEmpty()) {
            Toast.makeText(getContext(), "Please enter a rating", Toast.LENGTH_SHORT).show();
            return;
        }

        int rating;
        try {
            rating = Integer.parseInt(ratingStr);
            if (rating < 0 || rating > 100) {
                Toast.makeText(getContext(), "Rating must be 0-100", Toast.LENGTH_SHORT).show();
                return;
            }
        } catch (NumberFormatException e) {
            Toast.makeText(getContext(), "Invalid rating", Toast.LENGTH_SHORT).show();
            return;
        }

        if (caption.isEmpty()) {
            Toast.makeText(getContext(), "Caption is required", Toast.LENGTH_SHORT).show();
            return;
        }

        publishBtn.setEnabled(false);
        publishBtn.setText("Publishing...");

        String mediaType;
        switch (selectedMedia.mediaType) {
            case "MUSIC": mediaType = "MUSIC"; break;
            case "GAME":  mediaType = "GAME"; break;
            case "MOVIE": mediaType = "MOVIE"; break;
            case "SHOW":  mediaType = "MOVIE"; break;
            case "BOOK":  mediaType = "BOOK"; break;
            default:      mediaType = "MUSIC"; break;
        }

        // If external media not yet saved, save first
        if (selectedMedia.externalId != null && !selectedMedia.externalId.isEmpty() && selectedMedia.id == 0) {
            saveMediaThenPublish(rating, body, caption, mediaType);
            return;
        }

        createReview(rating, body, caption, mediaType);
    }

    private void createReview(int rating, String body, String caption, String mediaType) {
        Log.d("CreatePost", "Publishing: " + selectedMedia.title + " rating=" + rating + " id=" + selectedMedia.id);

        DTOmodels.review reviewRequest = new DTOmodels.review(mediaType, selectedMedia.title, rating, body);

        ApiClient.post(getContext(), "/reviews", reviewRequest, new Api_Interface() {
            @Override
            public void onSuccess(JSONObject response) {
                Log.d("CreatePost", "Review created, finding ID...");
                findReviewByTitle(selectedMedia.title, caption);
            }

            @Override
            public void onError(String message) {
                Log.e("CreatePost", "Review creation failed: " + message);
                publishFailed();
            }
        });
    }

    private void findReviewByTitle(String title, String caption) {
        ApiClient.getArray(getContext(), "/reviews", new Api_Array_Interface() {
            @Override
            public void onSuccess(JSONArray response) {
                try {
                    int reviewId = -1;
                    for (int i = response.length() - 1; i >= 0; i--) {
                        JSONObject obj = response.getJSONObject(i);
                        if (title.equals(obj.getString("title"))) {
                            reviewId = obj.getInt("id");
                            break;
                        }
                    }

                    if (reviewId == -1) {
                        Log.e("CreatePost", "Could not find created review");
                        publishFailed();
                        return;
                    }

                    Log.d("CreatePost", "Found review id: " + reviewId);
                    linkReviewToUser(reviewId, caption);
                } catch (Exception e) {
                    Log.e("CreatePost", "Error finding review: " + e.getMessage());
                    publishFailed();
                }
            }

            @Override
            public void onError(String message) {
                Log.e("CreatePost", "Failed to fetch reviews: " + message);
                publishFailed();
            }
        });
    }

                Log.d("CreatePost", "Review linked to user");
                linkReviewToMedia(reviewId, caption);
            }

            @Override
            public void onError(String message) {
                Log.e("CreatePost", "User link failed: " + message);
                publishPost(reviewId, caption);
            }
        });
    }

    private void linkReviewToMedia(int reviewId, String caption) {
        String endpoint = null;

        switch (selectedMedia.mediaType) {
            case "MUSIC":
                endpoint = "/reviews/" + reviewId + "/album/" + selectedMedia.id;
                break;
            case "GAME":
                endpoint = "/reviews/" + reviewId + "/game/" + selectedMedia.id;
                break;
            case "MOVIE":
                endpoint = "/reviews/" + reviewId + "/movie/" + selectedMedia.id;
                break;
            case "BOOK":
                endpoint = "/reviews/" + reviewId + "/book/" + selectedMedia.id;
                break;
        }

        if (endpoint == null) {
            publishPost(reviewId, caption);
            return;
        }

        Log.d("CreatePost", "Linking review to media: " + endpoint);

        ApiClient.putString(getContext(), endpoint, "", new Api_String_Interface() {
            @Override
            public void onSuccess(String response) {
                Log.d("CreatePost", "Review linked to media");
                publishPost(reviewId, caption);
            }

            @Override
            public void onError(String message) {
                Log.e("CreatePost", "Media link failed: " + message);
                publishPost(reviewId, caption);
            }
        });
    }

    private void publishPost(int reviewId, String caption) {
        String endpoint = "/reviews/" + reviewId + "/publish?authorId=" + currentUserId;
        DTOmodels.PostPublishRequest request = new DTOmodels.PostPublishRequest(caption, "PUBLIC");

        ApiClient.post(getContext(), endpoint, request, new Api_Interface() {
            @Override
            public void onSuccess(JSONObject response) {
                Log.d("CreatePost", "Post published!");
                updateMediaRating();
                if (getContext() != null) {
                    Toast.makeText(getContext(), "Post published!", Toast.LENGTH_SHORT).show();
                }
                if (publishListener != null) {
                    publishListener.onPostPublished();
                }
                dismiss();
            }

            @Override
            public void onError(String message) {
                Log.e("CreatePost", "Publish failed: " + message);
                publishFailed();
            }
        });
    }

    private void publishFailed() {
        publishBtn.setEnabled(true);
        publishBtn.setText("Publish");
        if (getContext() != null) {
            Toast.makeText(getContext(), "Failed to publish", Toast.LENGTH_SHORT).show();
        }
    }

    private void saveMediaThenPublish(int rating, String body, String caption, String mediaType) {
        String saveEndpoint = null;

        switch (selectedMedia.mediaType) {
            case "GAME":
                saveEndpoint = "/games/save?rawgId=" + selectedMedia.externalId;
                break;
            case "MOVIE":
                saveEndpoint = "/movies/save?tmdbId=" + selectedMedia.externalId + "&type=movie";
                break;
            case "SHOW":
                saveEndpoint = "/movies/save?tmdbId=" + selectedMedia.externalId + "&type=tv";
                break;
            case "BOOK":
                saveEndpoint = "/books/save?hardcoverId=" + selectedMedia.externalId;
                break;
            default:
                // Albums and artists auto-save, shouldn't reach here
                Log.e("CreatePost", "Unknown media type for save: " + selectedMedia.mediaType);
                publishFailed();
                return;
        }

        Log.d("CreatePost", "Saving media to database: " + saveEndpoint);

        ApiClient.post(getContext(), saveEndpoint, null, new Api_Interface() {
            @Override
            public void onSuccess(JSONObject response) {
                try {
                    // Get the database ID from the saved response
                    long savedId = response.optLong("id", 0);
                    if (savedId == 0) savedId = response.optLong("gameId", 0);
                    if (savedId == 0) savedId = response.optLong("movieId", 0);
                    if (savedId == 0) savedId = response.optLong("bookId", 0);

                    selectedMedia.id = savedId;
                    Log.d("CreatePost", "Media saved with id: " + savedId);

                    // Now create the review with the real database ID
                    createReview(rating, body, caption, mediaType);
                } catch (Exception e) {
                    Log.e("CreatePost", "Error parsing save response: " + e.getMessage());
                    publishFailed();
                }
            }

            @Override
            public void onError(String message) {
                Log.e("CreatePost", "Save media failed: " + message);
                publishFailed();
            }
        });
    }

    private void updateMediaRating() {
        String endpoint = null;

        switch (selectedMedia.mediaType) {
            case "MUSIC":
                endpoint = "/album/update-average-rating/" + selectedMedia.id;
                break;
            case "GAME":
                endpoint = "/games/" + selectedMedia.id + "/update-rating";
                break;
            case "MOVIE":
                endpoint = "/movies/" + selectedMedia.id + "/update-rating";
                break;
            case "BOOK":
                endpoint = "/books/" + selectedMedia.id + "/update-rating";
                break;
        }

        if (endpoint == null) return;

        Log.d("CreatePost", "Updating media rating: " + endpoint);

        ApiClient.putString(getContext(), endpoint, "", new Api_String_Interface() {
            @Override
            public void onSuccess(String response) {
                Log.d("CreatePost", "Media rating updated");
            }

            @Override
            public void onError(String message) {
                Log.e("CreatePost", "Rating update failed: " + message);
            }
        });
    }
