package com.example.androidexample;

import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;
import android.util.Log;
import android.view.View;
import android.view.inputmethod.EditorInfo;
import android.widget.Button;
import android.widget.EditText;
import android.widget.ImageView;
import android.widget.TextView;
import android.widget.Toast;

import androidx.appcompat.app.AppCompatActivity;

import com.android.volley.AuthFailureError;
import com.android.volley.Request;
import com.android.volley.Response;
import com.android.volley.VolleyError;
import com.android.volley.toolbox.JsonObjectRequest;
import com.android.volley.toolbox.StringRequest;
import com.google.gson.Gson;

import org.json.JSONException;
import org.json.JSONObject;

import java.util.HashMap;
import java.util.Map;

/**
 * @author Edwin Cepeda
 */
public class BookActivity extends AppCompatActivity {

    private ImageView coverImg;
    private View bookView;

    private TextView bookTxt, authorTxt, publishDateTxt, ratingTxt;
    private Button bookButton, authorButton;
    private EditText authorEdt;

    private String titleNeeded = null;
    private String authorNeeded = null;
    private int ID = 0;
    private JSONObject foundBook = null;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_book);

        Bundle extras = getIntent().getExtras();
        titleNeeded = null;
        authorNeeded = null;
        if (extras != null) {
            titleNeeded = extras.getString("itemName");
            authorNeeded = extras.getString("author");

        }
        bookView = findViewById(R.id.book_view);

        coverImg = findViewById(R.id.cover_img);
        bookTxt = findViewById(R.id.title_txt);
        authorTxt = findViewById(R.id.author_txt);
        publishDateTxt = findViewById(R.id.publishing_date_txt);
        ratingTxt = findViewById(R.id.rating_txt);
        bookButton = findViewById(R.id.delete_book_btn);
        authorEdt = findViewById(R.id.fix_author_edt);
        authorButton = findViewById(R.id.fix_author_btn);


        if (titleNeeded != null) {
            bookSearchRepository(titleNeeded);
        }

        bookButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                deleteBook(titleNeeded);
            }
        });

        authorButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String fixedAuthor = authorEdt.getText().toString();
                if (fixedAuthor == null || fixedAuthor.equals("")) {
                    Toast.makeText(getApplicationContext(), "No author written.", Toast.LENGTH_LONG).show();
                    return;
                }
                updateAuthor(fixedAuthor);
            }
        });


    }

    /**
     * This method searches our database for the requested book coming from the Search activity.
     * At this point, the book will always be in the repository, as that is handled in SearchActivity.
     *
     * @param mediaTitle The name of the book to be located in the Books repository.
     */
    private void bookSearchRepository(String mediaTitle) {
        String url = String.format("http://coms-3090-017.class.las.iastate.edu:8080/book/name?title=%s", mediaTitle);
        JsonObjectRequest retrieveBook = new JsonObjectRequest(Request.Method.GET, // HTTP method (GET request)
                url, // API URL
                null, // Request body (null for GET request)
                response -> {
                    //extract information out of returned JSON book object
                    String title;
                    String year;
                    String coverURL;
                    foundBook = response;
                    try {
                        Log.d("Server response:", response.toString());
                        title = foundBook.get("title").toString();
                        year = foundBook.get("publishedDate").toString();
                        coverURL = foundBook.get("thumbnailUrl").toString();
                        ID = (int) foundBook.get("id");
                    } catch (JSONException e) {
                        throw new RuntimeException(e);
                    }
                    //display results page for books
                    bookTxt.setText(title);
                    authorTxt.setText(authorNeeded);
                    publishDateTxt.setText("(" + year + ")");

                    //Credit for this image loading thing also goes to Cristian
                    if (coverURL == null || coverURL.isEmpty()) {
                        coverImg.setImageResource(R.drawable.error_image);
                        return;
                    }

                    VolleySingleton.getInstance(BookActivity.this).getImageLoader().get(coverURL,
                            com.android.volley.toolbox.ImageLoader.getImageListener(
                                    coverImg,
                                    R.drawable.cyval_logo,
                                    R.drawable.error_image
                            )
                    );
                },


                error -> {
                    int errorCode = error.networkResponse.statusCode;

                    if (errorCode == 404) { //If book could not be found in repository (404 error), use API search (OpenLibrary)
                        Toast.makeText(getApplicationContext(), "Could not retrieve book from database.", Toast.LENGTH_LONG).show();
                    } else {
                        Toast.makeText(getApplicationContext(), "Unknown error.", Toast.LENGTH_LONG).show();
                    }

                }
        ) {
            @Override
            public Map<String, String> getHeaders() throws AuthFailureError {
                // Define headers if needed
                HashMap<String, String> headers = new HashMap<>();
                return headers;
            }

            @Override
            protected Map<String, String> getParams() {
                // Define parameters if needed
                Map<String, String> params = new HashMap<>();
                // Example parameters (uncomment if needed)
                // params.put("param1", "value1");
                // params.put("param2", "value2");
                return params;
            }
        };

        // Adding request to the Volley request queue
        VolleySingleton.getInstance(getApplicationContext()).addToRequestQueue(retrieveBook);
    }

    /**
     * This method deletes the book currently being viewed. Restricted to Administrators.
     *
     * @param mediaTitle The name of the book to be deleted.
     */
    private void deleteBook(String mediaTitle) {
        String url = String.format("http://coms-3090-017.class.las.iastate.edu:8080/books/%d", ID);
        StringRequest deleteBook = new StringRequest(Request.Method.DELETE, // HTTP method (DELETE request)
                url, // API URL
                new Response.Listener<String>() {

                    @Override
                    public void onResponse(String response) {
                        Toast.makeText(getApplicationContext(), response, Toast.LENGTH_LONG).show();
                        Intent intent = new Intent(BookActivity.this, SearchActivity.class);
                        startActivity(intent);
                    }
                },
                new Response.ErrorListener() {

                    @Override
                    public void onErrorResponse(VolleyError error) {
                        Toast.makeText(getApplicationContext(), "Could not find book to delete", Toast.LENGTH_LONG).show();
                    }

                }
        ) {
            @Override
            public Map<String, String> getHeaders() throws AuthFailureError {
                // Define headers if needed
                HashMap<String, String> headers = new HashMap<>();
                return headers;
            }

            @Override
            protected Map<String, String> getParams() {
                // Define parameters if needed
                Map<String, String> params = new HashMap<>();
                // Example parameters (uncomment if needed)
                // params.put("param1", "value1");
                // params.put("param2", "value2");
                return params;
            }
        };

        // Adding request to the Volley request queue
        VolleySingleton.getInstance(getApplicationContext()).addToRequestQueue(deleteBook);
    }

    /**
     * If the author of a book is incorrect, this method can be used to update the current book
     * being viewed's author reference in the Books repository.
     *
     * @param newAuthor The name of the updated author reference.
     */
    private void updateAuthor(String newAuthor) {
        //update book information in foundBook
        try {
            foundBook.put("authors", newAuthor);
        } catch (JSONException e) {
            throw new RuntimeException(e);
        }
        String url = String.format("http://coms-3090-017.class.las.iastate.edu:8080/books/%d", ID);
        JsonObjectRequest update = new JsonObjectRequest(Request.Method.PUT, // HTTP method (GET request)
                url, // API URL
                foundBook, // the updated book to updated
                response -> {
                    Toast.makeText(getApplicationContext(), "Thank you for fixing!", Toast.LENGTH_LONG).show();

                    //update results page
                    String fixedText;
                    try {
                        fixedText = response.get("authors").toString();
                    } catch (JSONException e) {
                        throw new RuntimeException(e);
                    }
                    authorTxt.setText(fixedText);
                },


                error -> {
                    int errorCode = error.networkResponse.statusCode;

                    if (errorCode == 404) { //If book could not be found in repository (404 error), use API search (OpenLibrary)
                        Toast.makeText(getApplicationContext(), "Could not update book in database.", Toast.LENGTH_LONG).show();
                    } else {
                        Toast.makeText(getApplicationContext(), "Unknown error.", Toast.LENGTH_LONG).show();
                    }

                }
        ) {
            @Override
            public Map<String, String> getHeaders() throws AuthFailureError {
                // Define headers if needed
                HashMap<String, String> headers = new HashMap<>();
                return headers;
            }

            @Override
            protected Map<String, String> getParams() {
                // Define parameters if needed
                Map<String, String> params = new HashMap<>();
                // Example parameters (uncomment if needed)
                // params.put("param1", "value1");
                // params.put("param2", "value2");
                return params;
            }
        };

        // Adding request to the Volley request queue
        VolleySingleton.getInstance(getApplicationContext()).addToRequestQueue(update);
    }
}
