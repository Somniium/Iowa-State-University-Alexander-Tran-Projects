package com.example.androidexample;

import androidx.appcompat.app.AppCompatActivity;

import android.net.Uri;
import android.os.Bundle;
import android.view.View;
import android.view.inputmethod.EditorInfo;
import android.widget.Button;
import android.widget.EditText;
import android.widget.ImageView;
import android.widget.TextView;
import android.widget.Toast;

import com.google.gson.Gson;

import org.json.JSONArray;
import org.json.JSONObject;

/**
 * @author Cristian Alvarez
 * @version 1.0
 * An Android Activity class responsible for searching, displaying, and managing music albums.
 * This class handles UI interactions for finding albums, which first checks the remote
 * VM database and falls back to a Spotify API call if the album is not found locally.
 * It also supports submitting user ratings (which are linked to the album to calculate an average),
 * and deleting albums along with all their associated reviews from the database via the ApiClient.
 */
public class AlbumActivity extends AppCompatActivity {

    private ImageView coverImg;
    private View albumView;
    private EditText searchBar, ratingInput;
    private TextView albumTxt, artistTxt, releaseDateTxt, genreTxt, ratingTxt, tempTxt;
    private Button submitScoreBtn, deleteAlbumBtn;
    private DTOmodels.album currAlbum = null;

    /**
     * Initializes the activity, binds the UI components to their respective views,
     * and sets up the event listeners for the search bar and interactive buttons.
     *
     * @param savedInstanceState If the activity is being re-initialized after previously
     *                           being shut down, this Bundle contains the data it most
     *                           recently supplied. Otherwise, it is null.
     */
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_album);

        searchBar = findViewById(R.id.search_bar);
        albumView = findViewById(R.id.album_view);
        tempTxt = findViewById(R.id.temp_txt);

        coverImg = findViewById(R.id.cover_img);
        albumTxt = findViewById(R.id.name_txt);
        artistTxt = findViewById(R.id.artist_txt);
        releaseDateTxt = findViewById(R.id.release_date_txt);
        genreTxt = findViewById(R.id.genre_txt);
        ratingTxt = findViewById(R.id.rating_txt);

        ratingInput = findViewById(R.id.review_rating_input);
        submitScoreBtn = findViewById(R.id.btn_submit_score);
        deleteAlbumBtn = findViewById(R.id.btn_delete_album);

        albumView.setVisibility(View.GONE);
        tempTxt.setVisibility(View.VISIBLE);
        tempTxt.setText("search for an album!");

        searchBar.setOnEditorActionListener((v, actionId, event) -> {
            if (actionId == EditorInfo.IME_ACTION_SEARCH) {
                // Pass true because the user actually typed and searched
                searchForAlbum(searchBar.getText().toString(), true);
                return true;
            }
            return false;
        });

        submitScoreBtn.setOnClickListener(v -> {
            if (currAlbum != null) {
                submitRating();
            }
        });

        deleteAlbumBtn.setOnClickListener(v -> {
            if (currAlbum != null) {
                deleteAlbum();
            }
        });
    }

    /**
     * Sends a GET request to the remote database to search for an album by its name.
     * If the album is found, it populates the UI with the album's details. If the album
     * is not found, it triggers a fallback call to search the Spotify API.
     *
     * @param query        The name of the album to search for.
     * @param isUserSearch A boolean flag indicating if this was triggered manually by the user
     *                     (to determine whether to show a Toast notification).
     */
    private void searchForAlbum(String query, boolean isUserSearch) {
        String encodeQuery = Uri.encode(query);
        String endPoint = "/album/name?name=" + encodeQuery;

        ApiClient.get(this, endPoint, new Api_Interface() {
            @Override
            public void onSuccess(JSONObject response) {
                DTOmodels.album album = new Gson().fromJson(response.toString(), DTOmodels.album.class);

                currAlbum = album;
                if (isUserSearch) {
                    Toast.makeText(AlbumActivity.this, "Pulled from VM Database", Toast.LENGTH_SHORT).show();
                }

                tempTxt.setVisibility(View.GONE);
                albumView.setVisibility(View.VISIBLE);

                albumTxt.setText(album.name);
                artistTxt.setText(album.artist);
                releaseDateTxt.setText(album.releaseDate);
                genreTxt.setText(album.genre);

                int rating = album.rating;
                if (rating == 0) {
                    ratingTxt.setBackgroundColor(android.graphics.Color.parseColor("#848482"));
                    ratingTxt.setText("-");
                } else if (rating >= 80) {
                    ratingTxt.setBackgroundColor(android.graphics.Color.parseColor("#00FF00"));
                    ratingTxt.setText(String.valueOf(rating));
                } else if (rating >= 60) {
                    ratingTxt.setBackgroundColor(android.graphics.Color.parseColor("#FFFF00"));
                    ratingTxt.setText(String.valueOf(rating));
                } else {
                    ratingTxt.setBackgroundColor(android.graphics.Color.parseColor("#FF0000"));
                    ratingTxt.setText(String.valueOf(rating));
                }

                ApiClient.loadImage(AlbumActivity.this, album.coverURL, coverImg);

                searchBar.setText("");
                searchBar.clearFocus();
            }

            @Override
            public void onError(String message) {
                spotifyCall(query, isUserSearch);
            }
        });
    }

    /**
     * A fallback method that searches for an album using the Spotify API and saves it
     * to the remote database if found. Upon success, it updates the UI with the new data.
     *
     * @param query        The name of the album to search for.
     * @param isUserSearch A boolean flag indicating if this was triggered manually by the user.
     */
    public void spotifyCall(String query, boolean isUserSearch) {
        String encodeQuery = Uri.encode(query);
        String endPoint = "/search-albums-and-save?album_name=" + encodeQuery;

        ApiClient.get(this, endPoint, new Api_Interface() {
            @Override
            public void onSuccess(JSONObject response) {
                DTOmodels.album album = new Gson().fromJson(response.toString(), DTOmodels.album.class);

                currAlbum = album;

                // Only show the Toast if this was a manual search
                if (isUserSearch) {
                    Toast.makeText(AlbumActivity.this, "Pulled from Spotify", Toast.LENGTH_SHORT).show();
                }

                tempTxt.setVisibility(View.GONE);
                albumView.setVisibility(View.VISIBLE);

                albumTxt.setText(album.name);
                artistTxt.setText(album.artist);
                releaseDateTxt.setText(album.releaseDate);
                genreTxt.setText(album.genre);

                int rating = album.rating;
                if (rating == 0) {
                    ratingTxt.setBackgroundColor(android.graphics.Color.parseColor("#848482"));
                    ratingTxt.setText("-");
                } else if (rating >= 80) {
                    ratingTxt.setBackgroundColor(android.graphics.Color.parseColor("#00FF00"));
                    ratingTxt.setText(String.valueOf(rating));
                } else if (rating >= 60) {
                    ratingTxt.setBackgroundColor(android.graphics.Color.parseColor("#FFFF00"));
                    ratingTxt.setText(String.valueOf(rating));
                } else {
                    ratingTxt.setBackgroundColor(android.graphics.Color.parseColor("#FF0000"));
                    ratingTxt.setText(String.valueOf(rating));
                }

                ApiClient.loadImage(AlbumActivity.this, album.coverURL, coverImg);

                searchBar.setText("");
                searchBar.clearFocus();
            }

            @Override
            public void onError(String message) {
                Toast.makeText(AlbumActivity.this, "Error: " + message, Toast.LENGTH_LONG).show();
            }
        });
    }

    /**
     * Reads the user's input from the rating field, validates it (ensuring it is between 0 and 100),
     * and sends a POST request to create a new review in the remote database. Upon success,
     * it proceeds to link the new review to the current album.
     */
    private void submitRating() {
        String ratingStr = ratingInput.getText().toString().trim();
        if (ratingStr.isEmpty()) {
            Toast.makeText(this, "Please enter a rating", Toast.LENGTH_SHORT).show();
            return;
        }

        int rating = Integer.parseInt(ratingStr);

        if (rating < 0 || rating > 100) {
            Toast.makeText(this, "rating must be between 0 and 100!", Toast.LENGTH_SHORT).show();
            return;
        }

        submitScoreBtn.setEnabled(false);

        String uniqueTitle = "UniqueReview_" + System.currentTimeMillis();
        String defaultBody = "Submitted empty review";

        DTOmodels.review newReview = new DTOmodels.review("MUSIC", uniqueTitle, rating, defaultBody);

        ApiClient.post(this, "/reviews", newReview, new Api_Interface() {
            @Override
            public void onSuccess(JSONObject response) {
                findReviewIdAndLink(uniqueTitle);
            }

            @Override
            public void onError(String message) {
                submitScoreBtn.setEnabled(true);
                Toast.makeText(AlbumActivity.this, "Failed to submit score.", Toast.LENGTH_SHORT).show();
            }
        });
    }

    /**
     * Fetches all reviews from the database to find the newly generated review by its unique title.
     * Once the matching ID is found, it triggers the process to link this review to the current album.
     *
     * @param targetTitle The unique title string generated for the newly created review.
     */
    private void findReviewIdAndLink(String targetTitle) {
        ApiClient.getArray(this, "/reviews", new Api_Array_Interface() {
            @Override
            public void onSuccess(JSONArray response) {
                try {
                    int foundReviewId = -1;
                    for (int i = response.length() - 1; i >= 0; i--) {
                        if (response.getJSONObject(i).getString("title").equals(targetTitle)) {
                            foundReviewId = response.getJSONObject(i).getInt("id");
                            break;
                        }
                    }

                    if (foundReviewId != -1) linkReviewToAlbum(foundReviewId);
                    else {
                        submitScoreBtn.setEnabled(true);
                        Toast.makeText(AlbumActivity.this, "Failed to locate score ID.", Toast.LENGTH_SHORT).show();
                    }
                } catch (Exception e) {
                    e.printStackTrace();
                }
            }

            @Override
            public void onError(String message) {
                submitScoreBtn.setEnabled(true);
            }
        });
    }

    /**
     * Sends a PUT request to the database to associate a specific review ID with the
     * currently displayed album.
     *
     * @param reviewId The database ID of the review to be linked.
     */
    private void linkReviewToAlbum(int reviewId) {
        String endpoint = "/reviews/" + reviewId + "/album/" + currAlbum.albumId;

        ApiClient.putString(this, endpoint, "", new Api_String_Interface() {
            @Override
            public void onSuccess(String response) {
                syncAverageRating();
            }

            @Override
            public void onError(String message) {
                submitScoreBtn.setEnabled(true);
            }
        });
    }

    /**
     * Triggers a backend process to recalculate and update the average rating for the
     * current album based on all linked reviews. Upon success, it silently refreshes
     * the UI to display the new average rating.
     */
    private void syncAverageRating() {
        String endpoint = "/album/update-average-rating/" + currAlbum.albumId;

        ApiClient.putString(this, endpoint, "", new Api_String_Interface() {
            @Override
            public void onSuccess(String response) {
                submitScoreBtn.setEnabled(true);
                ratingInput.setText("");
                Toast.makeText(AlbumActivity.this, "Score averaged successfully!", Toast.LENGTH_SHORT).show();
                searchForAlbum(currAlbum.name, false);
            }

            @Override
            public void onError(String message) {
                submitScoreBtn.setEnabled(true);
            }
        });
    }

    /**
     * Initiates the deletion process for the currently displayed album. It first checks
     * if the album has any associated reviews. If it does, it deletes all child reviews
     * iteratively before attempting to delete the parent album.
     */
    private void deleteAlbum() {
        deleteAlbumBtn.setEnabled(false);

        if (currAlbum.albumReviews == null || currAlbum.albumReviews.isEmpty()) {
            deleteAlbumFromDatabase();
            return;
        }

        Toast.makeText(this, "Deleting reviews first...", Toast.LENGTH_SHORT).show();

        final int[] completedDeletes = {0};
        int totalReviews = currAlbum.albumReviews.size();

        for (DTOmodels.review rev : currAlbum.albumReviews) {
            String reviewEndpoint = "/reviews/" + rev.id;

            ApiClient.deleteString(this, reviewEndpoint, new Api_String_Interface() {
                @Override
                public void onSuccess(String response) {
                    completedDeletes[0]++;
                    checkAndTriggerAlbumDelete(completedDeletes[0], totalReviews);
                }

                @Override
                public void onError(String message) {
                    completedDeletes[0]++;
                    checkAndTriggerAlbumDelete(completedDeletes[0], totalReviews);
                }
            });
        }
    }

    /**
     * A helper method used during the review deletion process. It tracks the progress of
     * asynchronous delete requests and triggers the final album deletion once all associated
     * reviews have been successfully removed.
     *
     * @param completed The number of reviews that have been processed (either successfully deleted or errored).
     * @param total     The total number of reviews associated with the album.
     */
    private void checkAndTriggerAlbumDelete(int completed, int total) {
        if (completed == total) {
            deleteAlbumFromDatabase();
        }
    }

    /**
     * Sends the final DELETE request to the server to completely remove the album from
     * the database. Upon success, it resets the UI to the default search state.
     */
    private void deleteAlbumFromDatabase() {
        String endpoint = "/album/" + currAlbum.albumId;
        ApiClient.delete(this, endpoint, new Api_Interface() {
            @Override
            public void onSuccess(JSONObject response) {
                deleteAlbumBtn.setEnabled(true);
                albumView.setVisibility(View.GONE);
                tempTxt.setVisibility(View.VISIBLE);
                tempTxt.setText("Album Deleted.");
                Toast.makeText(AlbumActivity.this, "Album fully removed!", Toast.LENGTH_SHORT).show();
            }

            @Override
            public void onError(String message) {
                deleteAlbumBtn.setEnabled(true);
                Toast.makeText(AlbumActivity.this, "Delete failed: " + message, Toast.LENGTH_SHORT).show();
            }
        });
    }
}
