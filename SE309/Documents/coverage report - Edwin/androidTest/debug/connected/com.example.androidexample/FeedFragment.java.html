                } else {

                }

    private synchronized void checkAndMerge() {
        pendingCalls--;
        if (pendingCalls > 0) return;

        Log.d("FeedFragment", "All calls done. Posts: " + postItems.size() + " Media: " + mediaItems.size());

        Collections.shuffle(mediaItems);

        List<DTOmodels.FeedItem> merged = new ArrayList<>();
        Random random = new Random();
        int mediaIndex = 0;

        for (int i = 0; i < postItems.size(); i++) {
            merged.add(postItems.get(i));

            if (mediaIndex < mediaItems.size() && random.nextInt(3) == 0) {
                merged.add(mediaItems.get(mediaIndex));
                mediaIndex++;
            }
        }

        while (mediaIndex < mediaItems.size()) {
            merged.add(mediaItems.get(mediaIndex));
            mediaIndex++;
        }

        if (getActivity() != null) {
            getActivity().runOnUiThread(() -> {
                allItems.clear();
                allItems.addAll(merged);
                applyFilters();
                if (swipeRefresh != null) {
                    swipeRefresh.setRefreshing(false);
                }
            });
        }

        Log.d("FeedFragment", "Feed merged: " + allItems.size() + " total items");
    }

    private void applyFilters() {
        List<DTOmodels.FeedItem> filtered = new ArrayList<>();

        for (DTOmodels.FeedItem item : allItems) {
            if (currentViewType.equals("POSTS") && item.type != DTOmodels.FeedItem.TYPE_POST)
                continue;
            if (currentViewType.equals("MEDIA") && item.type != DTOmodels.FeedItem.TYPE_MEDIA)
                continue;

            if (!currentFilter.equals("ALL")) {
                if (item.type == DTOmodels.FeedItem.TYPE_POST && item.post != null) {
                    if (!currentFilter.equals(item.post.mediaType)) continue;
                } else if (item.type == DTOmodels.FeedItem.TYPE_MEDIA && item.media != null) {
                    if (!currentFilter.equals(item.media.mediaType)) continue;
                }
            }

            filtered.add(item);
        }

        feedItems.clear();
        feedItems.addAll(filtered);
        adapter.notifyDataSetChanged();

        if (filtered.isEmpty() && getContext() != null) {
            Toast.makeText(getContext(), "No results for this filter", Toast.LENGTH_SHORT).show();
        }
    }

    private void searchFeed(String query) {
        List<DTOmodels.FeedItem> filtered = new ArrayList<>();
        String lower = query.toLowerCase();

        for (DTOmodels.FeedItem item : allItems) {
            if (currentViewType.equals("POSTS") && item.type != DTOmodels.FeedItem.TYPE_POST)
                continue;
            if (currentViewType.equals("MEDIA") && item.type != DTOmodels.FeedItem.TYPE_MEDIA)
                continue;

            if (!currentFilter.equals("ALL")) {
                if (item.type == DTOmodels.FeedItem.TYPE_POST && item.post != null) {
                    if (!currentFilter.equals(item.post.mediaType)) continue;
                } else if (item.type == DTOmodels.FeedItem.TYPE_MEDIA && item.media != null) {
                    if (!currentFilter.equals(item.media.mediaType)) continue;
                }
            }

            if (item.type == DTOmodels.FeedItem.TYPE_POST && item.post != null) {
                if (item.post.mediaTitle != null && item.post.mediaTitle.toLowerCase().contains(lower)) {
                    filtered.add(item);
                }
            } else if (item.type == DTOmodels.FeedItem.TYPE_MEDIA && item.media != null) {
                if (item.media.title != null && item.media.title.toLowerCase().contains(lower)) {
                    filtered.add(item);
                }
            }
        }

        feedItems.clear();
        feedItems.addAll(filtered);
        adapter.notifyDataSetChanged();
        searchBar.clearFocus();

        if (filtered.isEmpty()) {
            Log.d("FeedFragment", "No local results for '" + query + "', searching external APIs...");
            if (getContext() != null) {
                Toast.makeText(getContext(), "Searching for '" + query + "'...", Toast.LENGTH_SHORT).show();
            }
            searchExternalApis(query);
        }
    }

    private void searchExternalApis(String query) {
        final List<DTOmodels.FeedItem> results = new ArrayList<>();
        int searchCount = 0;

        // Count how many searches we'll make based on filters
        if (currentFilter.equals("ALL") || currentFilter.equals("ALBUM")) searchCount++;
        if (currentFilter.equals("ALL") || currentFilter.equals("GAME")) searchCount++;
        if (currentFilter.equals("ALL") || currentFilter.equals("MOVIE")) searchCount++;
        if (currentFilter.equals("ALL") || currentFilter.equals("BOOK")) searchCount++;
        if (currentFilter.equals("ALL") || currentFilter.equals("ARTIST")) searchCount++;

        if (searchCount == 0) {
            Toast.makeText(getContext(), "No results found", Toast.LENGTH_SHORT).show();
            return;
        }

        final int[] pending = {searchCount};

        if (currentFilter.equals("ALL") || currentFilter.equals("ALBUM")) {
            ApiClient.get(getContext(), "/search-albums-and-save?album_name=" + query, new Api_Interface() {
                @Override
                public void onSuccess(JSONObject response) {
                    try {
                        DTOmodels.album album = new Gson().fromJson(response.toString(), DTOmodels.album.class);
                        synchronized (results) {
                            results.add(DTOmodels.FeedItem.fromAlbum(album));
                        }
                        Log.d("FeedFragment", "Found album: " + album.name);
                    } catch (Exception e) {
                        Log.e("FeedFragment", "Album search parse error: " + e.getMessage());
                    }
                    checkExternalDone(pending, results);
                }

                @Override
                public void onError(String message) {
                    Log.d("FeedFragment", "No album found: " + message);
                    checkExternalDone(pending, results);
                }
            });
        }

        if (currentFilter.equals("ALL") || currentFilter.equals("GAME")) {
            ApiClient.getArray(getContext(), "/search-games?q=" + query, new Api_Array_Interface() {
                @Override
                public void onSuccess(JSONArray response) {
                    try {
                        Gson gson = new Gson();
                        for (int i = 0; i < Math.min(response.length(), 3); i++) {
                            JSONObject obj = response.getJSONObject(i);

    private synchronized void checkExternalDone(int[] pending, List<DTOmodels.FeedItem> results) {
        pending[0]--;
        if (pending[0] > 0) return;

        if (getActivity() == null) return;

        getActivity().runOnUiThread(() -> {
            if (!results.isEmpty()) {
                allItems.addAll(results);
                mediaItems.addAll(results);

                feedItems.clear();
                feedItems.addAll(results);
                adapter.notifyDataSetChanged();
                searchBar.clearFocus();

                Toast.makeText(getContext(), "Found " + results.size() + " results! Added to database.", Toast.LENGTH_SHORT).show();
            } else {
                Toast.makeText(getContext(), "No results found", Toast.LENGTH_SHORT).show();
            }
        });
    }
